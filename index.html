import webbrowser

# 1. Writing to a file
with open("data.txt", "w") as file:
    file.write("Welcome to my profile sir.\n")
    file.write("At here you can view my Projects.\n")
    file.write("https://github.com/... \n")

# 2. Reading from a file
with open("data.txt", "r") as file:
    content = file.read()
    print(content)

# 3. Open to a website
print("\nOpening my github Profile...")
webbrowser.open("https://github.com/...")
